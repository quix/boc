enable(klass, method_name)

Enable Boc.value for the given instance method. 

class A
  def f
    p eval("x", Boc.value)
  end
  
  def self.g
    p eval("x", Boc.value)
  end
end

Boc.enable A, :f
Boc.enable A.singleton_class, :g

x = 33
A.new.f  # => 33
A.g  # => 33
